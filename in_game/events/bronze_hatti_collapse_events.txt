bronze_hatti_collapse.20 = {
	type = country_event
	title = bronze_hatti_collapse.20.title
	desc = bronze_hatti_collapse.20.desc
	category = situation_event
	image = "gfx/interface/illustrations/situation/SeaPeople.dds"
	outcome = negative
	orphan = yes

	trigger = {
		bronze_hatti_collapse_active_country = yes
		bronze_hatti_collapse_phase_at_least = { value = 2 }
		any_port_in_country = { is_coastal = yes }
	}

	immediate = {
		random_port_in_country = {
			limit = {
				is_coastal = yes
				NOT = { has_location_modifier = recently_raided_by_horde }
			}
			save_scope_as = target_location
		}
		if = {
			limit = { NOT = { exists = scope:target_location } }
			random_port_in_country = {
				limit = { is_coastal = yes }
				save_scope_as = target_location
			}
		}
	}

	option = {
		name = bronze_hatti_collapse.20.a
		custom_tooltip = bronze_hatti_collapse_20a_tt
		random_list = {
			25 = {
				trigger = { has_variable = bronze_hatti_coast_fortified }
				bronze_hatti_collapse_add_pressure = { value = -3 }
				custom_tooltip = bronze_hatti_collapse_raid_repulsed_tt
			}
			15 = {
				trigger = {
					scope:target_location ?= { modifier:fort_level > 0 }
				}
				bronze_hatti_collapse_add_pressure = { value = -2 }
				custom_tooltip = bronze_hatti_collapse_raid_repulsed_tt
			}
			15 = {
				trigger = {
					any_army = {
						count > 0
						unit_location = { is_coastal = yes }
					}
				}
				bronze_hatti_collapse_add_pressure = { value = -2 }
				custom_tooltip = bronze_hatti_collapse_raid_repulsed_tt
			}
			15 = {
				trigger = {
					OR = {
						navy_size_percentage >= 0.25
						any_navy = { count > 0 }
					}
				}
				bronze_hatti_collapse_add_pressure = { value = -2 }
				custom_tooltip = bronze_hatti_collapse_raid_repulsed_tt
			}
			70 = {
				bronze_hatti_collapse_local_raid_damage = yes
			}
		}
	}
}
