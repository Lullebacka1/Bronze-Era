# Deterministic successor map for the final phase of the Hatti collapse.
# Only locations directly owned by Hatti are eligible. Lost or foreign lands
# are never seized, which keeps this compatible with divergent campaigns.

bronze_hatti_collapse_materialize_successor = {
	if = {
		limit = {
			any_owned_location = {
				is_core_of = c:$tag$
			}
		}
		if = {
			limit = { country_exists = c:$tag$ }
			every_owned_location = {
				limit = { is_core_of = c:$tag$ }
				change_location_owner = c:$tag$
			}
		}
		else = {
			create_country_from_cores_in_our_locations = c:$tag$
		}
		c:$tag$ ?= {
			change_culture = culture:$culture$
			change_religion = religion:$religion$
			add_accepted_culture = culture:$accepted_culture$
			set_country_rank = country_rank:rank_kingdom
			set_variable = { name = bronze_hatti_successor_state value = 1 }
		}
	}
}

bronze_hatti_collapse_set_successor_capital = {
	if = {
		limit = {
			c:$tag$ ?= {
				owns = location:$location$
			}
		}
		c:$tag$ = {
			set_capital = location:$location$
		}
	}
}

bronze_hatti_collapse_mark_neo_hittite = {
	c:$tag$ ?= {
		set_variable = { name = bronze_hatti_neo_hittite_state value = 1 }
		set_variable = { name = bronze_hatti_religious_continuity value = 1 }
	}
}

bronze_hatti_collapse_convert_location_population = {
	if = {
		limit = { owns = location:$location$ }
		location:$location$ = {
			every_pop = {
				limit = {
					pop_size > 0.001
					OR = {
						NOT = { culture = culture:$culture$ }
						NOT = { religion = religion:$religion$ }
					}
				}
				split_pop = {
					fraction = $fraction$
					culture = culture:$culture$
					religion = religion:$religion$
				}
			}
		}
	}
}

bronze_hatti_collapse_preserve_royal_house = {
	ruler ?= {
		save_scope_as = bronze_hatti_fugitive_ruler
		dynasty ?= {
			save_scope_as = bronze_hatti_royal_dynasty
		}
	}
}

bronze_hatti_collapse_clear_successor_cores = {
	# Prevent legacy saves or earlier prototype events from expanding a
	# successor beyond the deterministic location list.
	every_owned_location = {
		if = { limit = { is_core_of = c:ARPAD } remove_core = c:ARPAD }
		if = { limit = { is_core_of = c:QUE } remove_core = c:QUE }
		if = { limit = { is_core_of = c:SAMAL } remove_core = c:SAMAL }
		if = { limit = { is_core_of = c:TILBA } remove_core = c:TILBA }
		if = { limit = { is_core_of = c:KUMMU } remove_core = c:KUMMU }
		if = { limit = { is_core_of = c:GURGU } remove_core = c:GURGU }
		if = { limit = { is_core_of = c:KHILA } remove_core = c:KHILA }
		if = { limit = { is_core_of = c:PATT } remove_core = c:PATT }
		if = { limit = { is_core_of = c:CARCH } remove_core = c:CARCH }
		if = { limit = { is_core_of = c:TABAL } remove_core = c:TABAL }
		if = { limit = { is_core_of = c:KAMMA } remove_core = c:KAMMA }
		if = { limit = { is_core_of = c:URATU } remove_core = c:URATU }
		if = { limit = { is_core_of = c:LUWIY } remove_core = c:LUWIY }
		if = { limit = { is_core_of = c:PHRYG } remove_core = c:PHRYG }
	}
}

bronze_hatti_collapse_establish_royal_house_in_que = {
	if = {
		limit = {
			country_exists = c:QUE
			exists = scope:bronze_hatti_fugitive_ruler
		}
		scope:bronze_hatti_fugitive_ruler = {
			move_country = c:QUE
			if = {
				limit = { is_married = yes }
				first_spouse = { move_country = c:QUE }
			}
			every_child = {
				limit = { is_alive = yes }
				move_country = c:QUE
			}
		}
		c:QUE = {
			set_new_ruler = scope:bronze_hatti_fugitive_ruler
		}
	}
	else_if = {
		limit = {
			country_exists = c:QUE
			exists = scope:bronze_hatti_royal_dynasty
		}
		c:QUE = {
			create_character = {
				culture = culture:hittite
				religion = religion:hittite
				dynasty = scope:bronze_hatti_royal_dynasty
				estate = estate_type:nobles_estate
				age = 35
				save_scope_as = bronze_hatti_que_ruler
			}
			set_new_ruler = scope:bronze_hatti_que_ruler
		}
	}
	else_if = {
		limit = { country_exists = c:QUE }
		c:QUE = {
			create_character = {
				culture = culture:hittite
				religion = religion:hittite
				estate = estate_type:nobles_estate
				age = 35
				save_scope_as = bronze_hatti_que_ruler
			}
			set_new_ruler = scope:bronze_hatti_que_ruler
		}
	}
	c:QUE ?= {
		set_variable = { name = bronze_hatti_legitimate_dynasty value = 1 years = 25 }
		add_legitimacy = legitimacy_mild_bonus
		add_prestige = prestige_mild_bonus
	}
}
