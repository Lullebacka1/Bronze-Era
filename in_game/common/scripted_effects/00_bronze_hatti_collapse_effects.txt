bronze_hatti_collapse_begin = {
	custom_description = {
		text = bronze_hatti_collapse_begin_text
		if = {
			limit = {
				situation:bronze_hatti_collapse = {
					situation_is_active = no
					situation_has_ended = no
				}
			}
			activate_situation = situation:bronze_hatti_collapse
		}
		set_variable = { name = bronze_hatti_collapse_active value = 1 }
		set_variable = { name = bronze_hatti_collapse_intro_seen value = 1 years = 80 }
		set_variable = { name = bronze_hatti_collapse_phase value = 1 }
		set_variable = { name = bronze_hatti_collapse_pressure value = 5 }
		set_variable = { name = bronze_hatti_famine_pressure value = 5 }
		set_variable = { name = bronze_hatti_vassal_oaths value = 4 }
	}
}

bronze_hatti_collapse_add_pressure = {
	custom_description = {
		text = bronze_hatti_collapse_add_pressure_text
		value = $value$
		if = {
			limit = { NOT = { has_variable = bronze_hatti_collapse_pressure } }
			set_variable = { name = bronze_hatti_collapse_pressure value = 0 }
		}
		change_variable = { name = bronze_hatti_collapse_pressure add = $value$ }
		if = {
			limit = { var:bronze_hatti_collapse_pressure > 100 }
			set_variable = { name = bronze_hatti_collapse_pressure value = 100 }
		}
		if = {
			limit = { var:bronze_hatti_collapse_pressure < 0 }
			set_variable = { name = bronze_hatti_collapse_pressure value = 0 }
		}
	}
}

bronze_hatti_collapse_sea_peoples_monthly_pressure = {
	custom_description = {
		text = bronze_hatti_collapse_sea_peoples_monthly_pressure_text
		# Minimal link to the separate Sea Peoples situation: Hatti only reads its global pressure.
		situation:sea_peoples_crisis ?= {
			if = {
				limit = { var:sea_peoples_global_phase >= 3 }
				c:HATTI ?= { bronze_hatti_collapse_add_pressure = { value = 1 } }
			}
			if = {
				limit = { var:sea_peoples_global_phase >= 4 }
				c:HATTI ?= {
					bronze_hatti_collapse_add_pressure = { value = 1 }
					bronze_hatti_collapse_add_famine = { value = 1 }
				}
			}
			if = {
				limit = { var:sea_peoples_global_pressure >= 70 }
				c:HATTI ?= { bronze_hatti_collapse_add_pressure = { value = 1 } }
			}
			if = {
				limit = { var:sea_peoples_global_famine_pressure >= 60 }
				c:HATTI ?= { bronze_hatti_collapse_add_famine = { value = 1 } }
			}
		}
	}
}

			create_country_from_cores_in_our_locations = c:$tag$

				change_culture = culture:$culture$
				change_religion = religion:$religion$
				add_accepted_culture = culture:$accepted_culture$

			}
		}
	}
}
