raid_on_anatolia_gui = {
	is_shown = { #Mycenaea Only mabey other greeks
		this = c:0001G #Mycenaea
		NOT = { has_global_variable = "trojan_war_phase_1_Ended" }
	}
	# what youll need to activate the effect of this action, e.g. manpower, treasury, cooldowns, etc.
	is_valid = {
		treasury >= 75
		manpower >= 250
		NOT = { has_variable = trojan_war_raid_on_anatolia_cooldown }
	}
	
	effect = {
		set_global_variable = { name = "trojan_war_raid_toggled" value = yes }
		set_variable = { name = "trojan_war_raid_on_anatolia_cooldown" 
		value = yes 
		duration = { months = 6 } # Duration 6 months coldown
	    } 
		random_list = {
			80 = {
				# to add more conties to affect the do like this modifer but under so you have modifer{} and then modifer{} so not in the same
				modifier = { 
					factor = 0.75
					c:Wilus = { has_country_modifier = "anatolia_coastline_is_defend" }
				}
				add_country_modifier = {
					name = "trojan_war_raid_on_anatolia_success"
					duration = {months = 12}
				}

			}
				#Rember to add here to if your adding an contry to the enmey side
			20 = {
				modifier = { 
					factor = 1.25
					c:Wilus = { has_country_modifier = "anatolia_coastline_is_defend" }
				}
				add_country_modifier = {
					name = "trojan_war_raid_on_anatolia_failed"
					duration = {months = 12}
				}
			}
		}
		add_manpower = -250
		add_treasury = -75
		remove_country_modifier = "trojan_war_raid_on_anatolia"
	}
}
