bronze_hatti_send_grain_to_cities = {
	type = situation
	show_message = yes
	ai_tick = monthly
	ai_tick_frequency = 18
	automation_tick = never
	automation_tick_frequency = 12

	price = price:bronze_hatti_send_grain_to_cities_price

	cooldown = {
		type = bronze_hatti_send_grain_to_cities
		years = 4
	}

	select_trigger = {
		looking_for_a = situation
		target_flag = recipient
		name = "choose_situation"
		column = { data = name }
		visible = {
			situation:bronze_hatti_collapse = this
			situation_is_active = yes
			scope:actor ?= { bronze_hatti_collapse_active_country = yes }
		}
	}
	potential = {
		scope:actor ?= {
			bronze_hatti_collapse_active_country = yes
			NOT = { has_variable = bronze_hatti_grain_relief_sent }
		}
	}
	allow = {
		scope:actor ?= { is_during_bankruptcy = no }
	}
	effect = {
		scope:actor ?= {
			set_variable = { name = bronze_hatti_grain_relief_sent value = 1 years = 4 }
			bronze_hatti_collapse_add_famine = { value = -14 }
			bronze_hatti_collapse_add_pressure = { value = -5 }
			add_estate_satisfaction = { type = estate_type:peasants_estate value = estate_satisfaction_weak_bonus }
			add_estate_satisfaction = { type = estate_type:nobles_estate value = estate_satisfaction_weak_penalty }
			custom_tooltip = bronze_hatti_send_grain_to_cities_tt
		}
	}
	ai_will_do = {
		add = 12
		subtract = 2
	}
}
