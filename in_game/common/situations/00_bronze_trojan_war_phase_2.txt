trojan_war_phase_2 = {
	# the war is connected with Bronze-Era\in_game\common\peace_treaties\00_bronze_trojan_war_phase_2_winner.txt
	monthly_spawn_chance = {
		value = 100
	}

	can_start = {
		current_date >= 14.1.1
		NOT = { has_global_variable = "trojan_war_phase_2_started" }
		NOT = { has_global_variable = "trojan_war_phase_2_ended" }
	}
	
	on_start = {
	        set_variable = { 
            name = situation_leader 
            value = c:0001G 
        }

        
        var:situation_leader = {
            trigger_event_silently = {
                id = start_trojan_war_phase_2.1
                days = 1
            }
        }
		set_global_variable = { name = "trojan_war_phase_2_started" value = yes }
		
		
		0001G = { add_country_modifier = { name = trojan_war_greece_modifier_phase_2 } }
		WILUS = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		AHHI = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		LYCIA = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		SEHA = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		LUKKA = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		MASA = { add_country_modifier = { name = trojan_war_anatolia_modifier_phase_2 } }
		
		# this one dosent trigger in game but when triggered in debugg gives wrong event
		# event located in \Bronze-Era\in_game\events\bronze_trojan_war_phase_2_events.txt
		# the event that triggers is in \Bronze-Era\in_game\events\bronze_trojan_war_phase_1_events.txt and is the trojan_war_raid_on_anatolia_success
	
    

		
	}

	can_end = {
    has_global_variable = "trojan_war_phase_2_started"
    NOT = { has_global_variable = "trojan_war_phase_2_ended" }
    
    
    current_date >= 14.6.1 
    
    NOT = {
        any_current_war = {
            any_war_participant = { tag = 0001G }
            any_war_participant = { tag = WILUS }
            cb_type = trojan_war_casus_belli
        }
    }
}

	end_conditions = {	
    0001G = {
        if = {
            limit = {
                has_global_variable = "greece_actually_won"
            }
            add_country_modifier = { name = trojan_war_greece_victory }
        }
        else = {
            	WILUS = { add_country_modifier = { name = trojan_war_anatolian_victory } }
				AHHI = { add_country_modifier = { name = trojan_war_anatolian_victory } }
				LYCIA = { add_country_modifier = { name = trojan_war_anatolian_victory } }
				SEHA = { add_country_modifier = { name = trojan_war_anatolian_victory } }
				LUKKA = { add_country_modifier = { name = trojan_war_anatolian_victory } }
				MASA = { add_country_modifier = { name = trojan_war_anatolian_victory } }
			}
		}
	}

	on_end = {
		set_global_variable = { name = "trojan_war_phase_2_ended" value = yes }
		0001G = {
            remove_country_modifier = trojan_war_greece_modifier_phase_2
        }
        WILUS = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
        AHHI = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
        LYCIA = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
        SEHA = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
        LUKKA = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
        MASA = {
            remove_country_modifier = trojan_war_anatolia_modifier_phase_2
        }
	}

	visible = {
		OR = {
			tag = 0001G # Mycenaea
			tag = WILUS # Wilusa
			tag = AHHI # Ahhiyawa
			tag = LYCIA # Lycia
			tag = SEHA # Seha River Land
			tag = LUKKA # Lukka
			tag = MASA # Masaa
		}
		NOT = { has_global_variable = "trojan_war_phase_2_ended"}
	}

}
