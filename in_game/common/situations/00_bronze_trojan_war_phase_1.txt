trojan_war_phase_1 = {
	monthly_spawn_chance = {
		value = 100
	}

	can_start = {
		current_date >= 5.1.1
		NOT = { has_global_variable = "trojan_war_phase_1_started" }
		NOT = { has_global_variable = "trojan_war_phase_1_ended" }
	}

	on_start = {
		set_global_variable = { name = "trojan_war_phase_1_started" value = yes }
	}

	can_end = {
		current_date >= 14.1.1
		has_global_variable = "trojan_war_phase_1_started"
		NOT = { has_global_variable = "trojan_war_phase_1_ended" }
	}

	end_conditions = {
		
	}

	on_end = {
		set_global_variable = { name = "trojan_war_phase_1_ended" value = yes }
		trigger_situation = {
			key = trojan_war_phase_2
			target = 0001G
		}
	}

	visible = {
		OR = {
			tag = 0001G # Mycenaea
			tag = WILUS # Wilusa
			tag = AHHI # Ahhiyawa
			tag = LYCIA # Lycia
			tag = SEHA # Seha River Land
			tag = LUKKA # Lukka
			tag = MASA # Masaa
		}
		NOT = { has_global_variable = "trojan_war_phase_1_ended" }
	}

	# EU5 situation panels are resolved from the situation key:
	# in_game/gui/panels/situation/trojan_war_phase_1.gui
}
