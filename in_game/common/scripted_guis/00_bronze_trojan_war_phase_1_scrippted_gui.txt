				}

					factor = 0.89

				}
				modifier = {
					factor = 0.89
					c:LUKKA = { has_country_modifier = anatolia_coastline_is_defend }
				}
				modifier = {
					factor = 0.89
					c:MASA = { has_country_modifier = anatolia_coastline_is_defend }
				}
			  		
				add_country_modifier = {
					modifier = trojan_war_raid_on_anatolia_success
					months = 12
				}
				trigger_event_non_silently = trojan_war_raid_on_anatolia_success.1 
			}

			20 = {
				modifier = {
					factor = 1.12

defend_coastline_gui = {
	scope = country

	is_shown = {
		OR = {
			tag = WILUS # Wilusa
			tag = AHHI # Ahhiyawa
			tag = LYCIA # Lycia
			tag = SEHA # Seha River Land
			tag = LUKKA # Lukka
			tag = MASA # Masaa
			}
		NOT = { has_global_variable = "trojan_war_phase_1_ended" }
	}

is_valid = {
		root = {
			treasury = { value >= 25 }
			manpower = { value >= 150 }
			NOT = { has_variable = trojan_war_defend_coastline_cooldown }
		}
	}

	effect = {
		set_global_variable = { name = "trojan_war_defend_toggled" value = yes }
		set_variable = {
			name = trojan_war_defend_coastline_cooldown
			value = yes
			months = 6
		}

		add_country_modifier = {
			modifier = anatolia_coastline_is_defend
			months = 6
		}
		add_manpower_pool = -150
		add_gold = -25
		trigger_event_non_silently = { id = anatolia_coastline_is_defend.1 }  
	}

	ai_is_valid = { always = no }
}
