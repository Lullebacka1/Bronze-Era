# Bronze Age metallurgy.  The building is constructed with vanilla workshop
# materials; copper and tin are economic inputs only after the foundry exists.
bronze_age_bronze_foundry = {
	is_foreign = no
	pop_type = laborers
	max_levels = workshop_max_level

	category = basic_industry_category
	employment_size = workshop_employment

	town = yes
	city = yes
	megalopolis = yes

	build_time = workshop_build_time
	construction_demand = workshop_construction

	unique_production_methods = {
		bronze_age_full_bronze_supply = {
			copper = 0.75
			tin = 0.08
			produced = bronze_age_bronze
			output = 1.0

			debug_max_profit = 0.35
			category = workshop_input
		}

		# Fallback for disrupted tin routes.  It keeps the foundry useful, but
		# the output is intentionally poor to make tin trade strategically vital.
		bronze_age_bronze_shortage = {
			copper = 0.55
			produced = bronze_age_bronze
			output = 0.25

			debug_max_profit = 0.08
			category = workshop_input
		}
	}

	custom_tags = { workshop bronze_age_industry }
}
